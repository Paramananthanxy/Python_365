import xmltodict
import json

with open ('exmaple.xml',encoding='utf-8') as xml_file: 

##encoding='utf-8 -->ensures correct character reading..
    data_dict = xmltodict.parse(xml_file.read())
##xml_file.read()  --> reading the entire xml file.
##xmltodict.parse  --> Converting the xml into python dictionary.
    json_data =json.dumps(data_dict, indent=4) -->storing json_data

with open("data.json".'w') as json_file:
json_file.write(json_data)
